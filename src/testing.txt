PARSER.UNIT [
	PARSER.VARASSIGN [
		LEXER.ID main,
		'=',
		PARSER.FUNCDECL [
			[
				'a',
				'b'
			],
			[
				PARSER.IF [
					PARSER.CONDITION [ // condition
						LEXER.ID a,
						LEXER.ID b,
						'=='
					],
					[ // if true
						PARSER.RETURN LEXER.ID a
					],
					[ // elseif chain
						PARSER.IF [
							PARSER.CONDITION [
								LEXER.ID a,
								LEXER.ID b,
								'=='
							],
							[
								PARSER.RETURN LEXER.ID a
							]
						],
						PARSER.IF [
							PARSER.CONDITION [
								LEXER.ID a,
								LEXER.ID b,
								'=='
							],
							[
								PARSER.RETURN LEXER.ID a
							]
						],
						PARSER.IF [
							PARSER.CONDITION [
								LEXER.ID a,
								LEXER.ID b,
								'=='
							],
							[
								PARSER.RETURN LEXER.ID a
							]
						]
					],
					[ // else
						PARSER.RETURN PARSER.EXPR [
							'-',
							LEXER.ID a,
							LEXER.ID b
						]
					]
				]
			]
		]
	]
]
